# This override merges with /opt/docker/shekel/docker-compose.yml (which is
# byte-identical to the repo's docker-compose.yml as of 2026-04-17).
#
# Two production-only changes:
#   1. shekel-prod-app joins the external 'homelab' network (in addition to
#      'backend'), making it reachable from cloudflared/nginx/jellyfin/immich/unifi.
#   2. The bundled shekel-prod-nginx service is placed in the 'disabled' profile,
#      so it does not start.  The shared /opt/docker/nginx container (defined in
#      /opt/docker/docker-compose.yml) is the only Nginx in this deployment.
#
# ---- BEGIN docker-compose.override.yml ----

# Use the central Nginx reverse proxy instead of the bundled one.
services:
  app:
    networks:
      - backend
      - homelab

  nginx:
    profiles: ["disabled"]

networks:
  homelab:
    external: true

# ---- END docker-compose.override.yml ----
