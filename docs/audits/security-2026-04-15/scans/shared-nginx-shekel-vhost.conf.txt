# /opt/docker/nginx/conf.d/shekel.conf
# Reverse proxy for Shekel budget app.
# LAN clients reach this via the OPNsense DNS override for
# shekel.saltyreformed.com, which resolves to this machine's LAN IP.

# Redirect HTTP to HTTPS
server {
    listen 80;
    server_name shekel.saltyreformed.com;
    return 301 https://$host$request_uri;
}

server {
    listen 443 ssl;
    http2 on;
    server_name shekel.saltyreformed.com;

    # Let's Encrypt certificates
    ssl_certificate /etc/nginx/certs/fullchain.pem;
    ssl_certificate_key /etc/nginx/certs/key.pem;
    ssl_protocols TLSv1.2 TLSv1.3;
    ssl_ciphers HIGH:!aNULL:!MD5;
    ssl_prefer_server_ciphers on;
    ssl_session_cache shared:SSL:10m;
    ssl_session_timeout 10m;

    # Shekel is a budget app with no large uploads.
    client_max_body_size 5M;

    location / {
        # 'shekel-prod-app' resolves via Docker DNS to the shekel app container
        # on the shared homelab network.  Gunicorn listens on port 8000.
        # Variable in proxy_pass forces runtime re-resolution (see
        # resolver directive in nginx.conf).
        set $upstream_shekel shekel-prod-app:8000;
        proxy_pass http://$upstream_shekel;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;

        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection $connection_upgrade;
        proxy_buffering off;
    }
}
