# /opt/docker/nginx/nginx.conf
# Main Nginx configuration for the homelab reverse proxy.
# Per-service server blocks live in /etc/nginx/conf.d/*.conf

user  nginx;
worker_processes  auto;

error_log  /var/log/nginx/error.log warn;
pid        /var/run/nginx.pid;

events {
    worker_connections  1024;
}

http {
    include       /etc/nginx/mime.types;
    default_type  application/octet-stream;

    log_format  main  '$remote_addr - $remote_user [$time_local] "$request" '
                      '$status $body_bytes_sent "$http_referer" '
                      '"$http_user_agent" "$http_x_forwarded_for"';

    access_log  /var/log/nginx/access.log  main;

    sendfile        on;
    tcp_nopush      on;
    tcp_nodelay     on;
    keepalive_timeout  65;
    types_hash_max_size 2048;

    # Do not reveal Nginx version in headers
    server_tokens off;

    # WebSocket upgrade mapping
    # If the client sends an Upgrade header, $connection_upgrade becomes
    # "upgrade". Otherwise it becomes "close". This is the standard way
    # to handle WebSocket proxying in Nginx.
    map $http_upgrade $connection_upgrade {
        default upgrade;
        ''      close;
    }

    # Use Docker's embedded DNS so upstreams are re-resolved at runtime
    # (instead of cached once at config load). Without this, a container
    # restart that changes its IP causes 502s until nginx is reloaded.
    # Paired with `set $upstream_x ...; proxy_pass http://$upstream_x;`
    # in each server block to force per-request resolution.
    resolver 127.0.0.11 valid=30s ipv6=off;

    # Include per-service server blocks
    include /etc/nginx/conf.d/*.conf;
}
