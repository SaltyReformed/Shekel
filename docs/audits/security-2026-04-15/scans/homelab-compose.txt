# /opt/docker/docker-compose.yml
# Homelab services: Jellyfin + Nginx reverse proxy + Cloudflare Tunnel
# All services share the 'homelab' bridge network for internal communication.
# Only Nginx exposes ports to the host (80/443 for LAN access).
# cloudflared makes outbound-only connections to Cloudflare (no published ports).
# Jellyfin has no published ports -- it is only reachable via Nginx (LAN) or
# cloudflared (remote).

services:

  jellyfin:
    image: jellyfin/jellyfin:latest
    container_name: jellyfin
    init: true
    restart: unless-stopped
    security_opt:
      - no-new-privileges:true
    cap_drop:
      - ALL
    cap_add:
      - DAC_OVERRIDE    # root must write to config files owned by host user
    healthcheck:
      test: ["CMD-SHELL", "bash -c 'exec 3<>/dev/tcp/127.0.0.1/8096 && echo -e \"GET /health HTTP/1.1\\r\\nHost: localhost\\r\\nConnection: close\\r\\n\\r\\n\" >&3 && grep -q \"200 OK\" <&3'"]
      interval: 30s
      timeout: 10s
      retries: 3
      start_period: 30s
    deploy:
      resources:
        limits:
          memory: 8G
        reservations:
          memory: 1G
    environment:
      - TZ=America/New_York
    volumes:
      - /opt/docker/jellyfin/config:/config
      - /opt/docker/jellyfin/cache:/cache
      - /mnt/movies:/media/movies:ro
      - /mnt/tv:/media/tv:ro
      - /mnt/multimedia:/media/multimedia:ro
      - /mnt/media:/media/media:ro
    tmpfs:
      - /config/transcodes:size=4G
      - /tmp:size=256M
    devices:
      - /dev/dri/renderD128:/dev/dri/renderD128
    networks:
      - homelab
    group_add:
      - "987"

  unifi:
    image: jacobalberty/unifi:latest
    container_name: unifi
    init: true
    restart: unless-stopped
    deploy:
      resources:
        limits:
          memory: 2G
        reservations:
          memory: 512M
    environment:
      - TZ=America/New_York
    volumes:
      - /opt/docker/unifi:/unifi
    ports:
      - "8080:8080"
      - "3478:3478/udp"
      - "10001:10001/udp"
    networks:
      - homelab

  nginx:
    image: nginx:latest
    container_name: nginx
    restart: unless-stopped
    security_opt:
      - no-new-privileges:true
    cap_drop:
      - ALL
    cap_add:
      - NET_BIND_SERVICE
      - CHOWN
      - DAC_OVERRIDE
      - SETUID
      - SETGID
    healthcheck:
      test: ["CMD-SHELL", "curl -f http://localhost:80/ || exit 1"]
      interval: 30s
      timeout: 5s
      retries: 3
      start_period: 10s
    deploy:
      resources:
        limits:
          memory: 256M
    ports:
      - "80:80"
      - "443:443"
    volumes:
      - /opt/docker/nginx/nginx.conf:/etc/nginx/nginx.conf:ro
      - /opt/docker/nginx/conf.d:/etc/nginx/conf.d:ro
      - /opt/docker/nginx/certs:/etc/nginx/certs:ro
      - /opt/docker/nginx/logs:/var/log/nginx
    depends_on:
      jellyfin:
        condition: service_healthy
      unifi:
        condition: service_healthy
    networks:
      - homelab

  cloudflared:
    image: cloudflare/cloudflared:latest
    container_name: cloudflared
    init: true
    restart: unless-stopped
    security_opt:
      - no-new-privileges:true
    cap_drop:
      - ALL
    command: tunnel --metrics 0.0.0.0:2000 run
    healthcheck:
      test: ["CMD", "cloudflared", "tunnel", "ready"]
      interval: 5s
      timeout: 5s
      retries: 3
      start_period: 15s
    deploy:
      resources:
        limits:
          memory: 256M
    volumes:
      - /opt/docker/cloudflared:/etc/cloudflared:ro
    depends_on:
      jellyfin:
        condition: service_healthy
    networks:
      - homelab

networks:
  homelab:
    name: homelab
    driver: bridge

# ---- END /opt/docker/docker-compose.yml ----
