# Polyglot-cleanup Phase 5 linters (pip-installable). The binary tools
# (shellcheck, shfmt, biome, actionlint, typos, gitleaks, hadolint) are
# installed on the host and pinned by download in CI; these four ship
# via pip, so they live here and CI installs them with this file.
# Versions match the host toolchain; pinning guards against rule/dictionary
# drift newly flagging unchanged files between releases. rumdl is the
# Markdown lint + format gate (living docs only; see .rumdl.toml excludes).
yamllint==1.38.0
zizmor==1.25.2
djlint==1.39.2
rumdl==0.2.14

# Pre-commit framework: runs the local hooks in .pre-commit-config.yaml
# (pylint app/ + the custom checkers) before each commit, mirroring the CI
# gate. A pre-merge belt-and-suspenders layer for commits made outside a
# Claude Code session; CI remains the enforced gate.
pre-commit==4.0.1

# Manual mobile / browser verification harness.  Drives a headless
# Chromium at iPhone-sized viewports against the local dev Flask
# server; the runnable scripts live under tests/manual/ and are
# operator-invoked, not pytest-collected (filename pattern is outside
# pytest.ini's python_files = test_*.py).  After ``pip install`` the
# Chromium binary still has to be fetched once with
# ``playwright install chromium`` (writes to ~/.cache/ms-playwright,
# ~150 MiB).  Used by tests/manual/save_dev_session.py and
# tests/manual/verify_mobile_grid_*.py.
playwright==1.60.0
